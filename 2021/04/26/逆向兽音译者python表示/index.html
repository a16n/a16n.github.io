java代码

package tk.dreamydragon.roartranslater;

public class TxtToBeast {
    private static final char[] bd = {22007, 21596, 21834, '~'};

    public static String toBeast(String tf) {
        return HexToBeast(ToHex(tf));
    }

    public static String fromBeast(String tf) {
        return FromHex(BeastToHex(tf));
    }

    public static void setBeastChars(char cf0, char cf1, char cf2, char cf3) {
        bd[0] = cf0;
        bd[1] = cf1;
        bd[2] = cf2;
        bd[3] = cf3;
    }

    public static char[] getBeastChars() {
        return new char[]{bd[0], bd[1], bd[2], bd[3]};
    }

    private static String ToHex(String gbString) {
        char[] utfBytes = gbString.toCharArray();
        StringBuffer unicodeBytes = new StringBuffer();
        for (char c : utfBytes) {
            String hexB = Integer.toHexString(c);
            while (hexB.length() < 4) {
                hexB = "0" + hexB;
            }
            unicodeBytes.append(hexB);
        }
        return unicodeBytes.toString();
    }

    private static String FromHex(String dataStr) {
        StringBuffer buffer = new StringBuffer();
        int start = 0;
        for (int end = 4; end <= dataStr.length(); end += 4) {
            buffer.append(new Character((char) Integer.parseInt(dataStr.substring(start, end), 16)).toString());
            start += 4;
        }
        return buffer.toString();
    }

    private static String HexToBeast(String tf) {
        char[] tfArray = tf.toCharArray();
        StringBuffer beast = new StringBuffer();
        for (int i = 0; i < tfArray.length; i++) {
            int k = Integer.valueOf(String.valueOf(tfArray[i]), 16).intValue() + (i % 16);
            if (k >= 16) {
                k -= 16;
            }
            beast.append("" + bd[k / 4] + bd[k % 4]);
        }
        return beast.toString();
    }

    private static String BeastToHex(String bf) {
        char[] bfArray = bf.toCharArray();
        StringBuffer out = new StringBuffer();
        for (int i = 0; i <= bf.length() - 2; i += 2) {
            int pos1 = 0;
            int pos2 = 0;
            char c = bfArray[i];
            while (pos1 <= 3 && c != bd[pos1]) {
                pos1++;
            }
            char c2 = bf.charAt(i + 1);
            while (pos2 <= 3 && c2 != bd[pos2]) {
                pos2++;
            }
            int k = ((pos1 * 4) + pos2) - ((i / 2) % 16);
            if (k < 0) {
                k += 16;
            }
            out.append(Integer.toHexString(k));
        }
        return out.toString();
    }
}

# 逆向兽音算法
ch = [22007, 21596, 21834, '~']
for i in range(len(ch)):
    if str(ch[i]).isnumeric():
        ch[i] = chr(ch[i])
print(ch)
# ####################################上面的代码为了恢复加密表['嗷', '呜', '啊', '~']
m = input("请输入原文:")


def ToHex(m):
    text = m
    hexstr = ''
    for i in range(len(text)):
        temphex = hex(ord(text[i])).replace('0x', '')
        while len(temphex) < 4:
            temphex = '0' + temphex
        hexstr += temphex
    return hexstr


def HexToBeast(temphex):
    beast = ''
    for i in range(len(temphex)):
        tempdec = int(temphex[i], 16) + i % 16  # 十六进制字符串转为int类型的十进制
        if tempdec >= 16:
            tempdec -= 16
        beast += ch[tempdec // 4] + ch[tempdec % 4]
    beast = '~呜嗷' + beast + '啊'
    return beast


c = HexToBeast(ToHex(m))
print('密文为:', c)
def BeastToText(c):
    text = c[3:len(c)-1]
    print(text)
    out = ''
    char = ''
    for i in range(0, len(text), 2):
        pos1, pos2 = 0, 0

        c = text[i]
        while pos1 <= 3 and c != ch[pos1]:
            pos1 += 1

        c2 = text[i + 1]
        while pos2 <= 3 and c2 != ch[pos2]:
            pos2 += 1

        temp = (pos1 * 4) + pos2 - ((i // 2) % 16)

        if temp < 0:
            temp += 16
        char += hex(temp).replace('0x', '')
        if len(char) == 4:
            for i in char:
                if i == '0':
                    char = char[1:]
                else:
                    break
            out += chr(int(char, 16))
            char = ''
    return out


jie = BeastToText(c)
print('解密后为:', jie)
