# -*- coding: utf-8 -*-#
# -------------------------------------------------------------------------------
# Name:         CrcCrash
# Description:  for txt file to crash crc
# Author:       M4XLMUM
# Date:         2021/3/7
# -------------------------------------------------------------------------------
import zipfile
import string
import binascii

dic = string.printable   #猜测文件内容的字典

def getCrc():
    zip = zipfile.ZipFile("flag.zip", "r")
    zip_list = zip.namelist()
    crclist = []
    for txt in zip_list:
        if '.txt' in txt:
            crclist.append(zip.getinfo(txt).CRC)
    return crclist


filecrc = getCrc()
res = ''
def calculateCrc():
    global res
    for filenum in range(9):
        for i in dic:    # 层数，本题中只有一个字节，那就用一层循环就行
            s = str(i).encode()
            if binascii.crc32(s) == filecrc[filenum]:
                res = res + i
                break
    return res


result = calculateCrc()
print(result)
