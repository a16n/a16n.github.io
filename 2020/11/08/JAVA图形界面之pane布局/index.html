基于JFrame的图形界面

最近在学JFrame，下面用这篇博客来记一下学习笔记，主要关于的是今天学到的比较印象深刻的代码及其中的细节吧。

pane面板布局及三种布局管理器

这里的三种布局管理器分别为流布局管理器FlowLayout、边布局管理器BorderLayout、网格布局管理器GridLayout

这种布局管理器怎样进行设置呢？就用下面的代码

this.getContentPane().setLayout(new GridLayout(2, 1));  //设置网格布局，其中为2行1列

其余的布局设置亦相似与上面的设置方法，下面解释一下各块的意义
this.getContentPane()   //代表主的内容面板，这里当然也可以是向下的各级内容面板
setLayout()             //设置布局的函数，是panel类所具有的方法。
new GridLayout(2, 1)    //表示创建了个两行一列的网格布局管理器。

panel套panel

这里主要用于分布复杂的软件，有上有下这样的，（目前只学了这种设置组件的位置的控制方式。
主的contentpane套pane

JPanel jp1 = new JPanel();           //新建一个内容面板jp1
this.getContentPane().add(jp1);      //将这个内容面板添加（add）到主的内容面板（this.getContentPane()）之内

其他的各级的pane相互套的方式与之类似。

Window2.java

package JFrameTest3;

import java.awt.BorderLayout;
import java.awt.Button;
import java.awt.Color;
import java.awt.GridLayout;

import javax.swing.JButton;
import javax.swing.JFrame;
import javax.swing.JPanel;

public class Window2 extends JFrame{
	public Window2() {
		//**下面为通用开头*/
		super("JFrame测试");
		this.setBounds(400, 400, 400, 400);
		this.getContentPane().setBackground(Color.gray);
		this.setDefaultCloseOperation(EXIT_ON_CLOSE);
		
		
		//**下面是网格布局（第一布局）*/
		this.getContentPane().setLayout(new GridLayout(2, 1));
		
		//**下面是上面的部分（即为第一步分开上面的部分）*/
		JPanel jp1 = new JPanel();
		this.getContentPane().add(jp1);
		jp1.setLayout(new BorderLayout());
		//**设置上面的左右（或者说东、西？）部分为按钮1、4*/
		jp1.add(new Button("this is button1"), "West");
		jp1.add(new Button("this is button4"), "East");
		//**中间的部分（这里设置为一个面板，再向面板之中添加其他组件）*/
		JPanel jp1_2 = new JPanel();
		//jp1_2加入到面板1中，并且放到中间
		jp1.add(jp1_2, "Center");     
		//设置为网格布局，并且向该面板之中添加两个按钮2、3
		jp1_2.setLayout(new GridLayout(2,1));
		jp1_2.add(new Button("this is button2"));
		jp1_2.add(new Button("this is button3"));
		
		
		//**添加下面的部分的面板*/
		JPanel jp2 = new JPanel();
		this.getContentPane().add(jp2);
		//设置为边布局管理器，并添加按钮5、10与面板的左右部分
		jp2.setLayout(new BorderLayout());
		jp2.add(new Button("this is button5"), "West");
		jp2.add(new Button("this is button10"), "East");
		
		//面板之中添加另一面板
		JPanel jp2_1 = new JPanel();
		//这里设置为网格布局
		jp2_1.setLayout(new GridLayout(2,2));
		//也放在中间
		jp2.add(jp2_1, "Center");
		//在网格布局管理器里面添加四个按钮6、7、8、9
		jp2_1.add(new Button("this is buuton6"));
		jp2_1.add(new Button("this is buuton7"));
		jp2_1.add(new Button("this is buuton8"));
		jp2_1.add(new Button("this is buuton9"));
		
		
		this.setVisible(true);
	}
}
